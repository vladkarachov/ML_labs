def plot_curves(classifier, Xtest, Ytest):
    plot_precision_recall_curve(classifier, Xtest, Ytest)
    plot_roc_curve(classifier, Xtest, Ytest)
    fpr, tpr, thresholds = roc_curve(Ytest, classifier.predict(Xtest))
    print("Auc: ", auc(fpr, tpr))

def draw_contour(model, X):
    h=0.01
    x_min=X[:,0].min()
    x_max=X[:,0].max()
    y_min=X[:,1].min()
    y_max=X[:,1].max()

    xx= np.meshgrid(np.arange(x_min, x_max, h),
                        np.arange(y_min,y_max, h))
    plt.scatter(X[:,0], X[:,1], c=Y, cmap=plt.cm.coolwarm)
    #plt.plot(svm_vectors[0])
    Z=model.predict(np.c_[xx[0].ravel(), xx[1].ravel()])
    Z=Z.reshape(xx[0].shape)
    plt.contourf(xx[0], xx[1], Z, cmap=plt.cm.coolwarm, alpha=0.8)
